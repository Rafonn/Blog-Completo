import java.util.ArrayList;
import java.util.List;
import java.util.Random;
import java.util.Scanner;

public class RandomCodeExample {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        Random random = new Random();

        // Gerando números aleatórios entre 1 e 100
        int randomNum = random.nextInt(100) + 1;
        System.out.println("Número aleatório gerado: " + randomNum);

        // Gerar uma string aleatória
        String randomString = generateRandomString(10);
        System.out.println("String aleatória gerada: " + randomString);

        // Adicionando números aleatórios em uma lista
        List<Integer> numbersList = new ArrayList<>();
        for (int i = 0; i < 10; i++) {
            numbersList.add(random.nextInt(100) + 1);
        }

        System.out.println("Lista de números aleatórios gerada:");
        for (int number : numbersList) {
            System.out.println(number);
        }

        // Perguntando ao usuário para adivinhar um número
        System.out.println("\nTente adivinhar o número aleatório gerado (entre 1 e 100):");
        int userGuess = scanner.nextInt();

        if (userGuess == randomNum) {
            System.out.println("Parabéns! Você adivinhou corretamente.");
        } else {
            System.out.println("Tente novamente. O número era: " + randomNum);
        }

        // Calculando a soma dos números na lista
        int sum = 0;
        for (int number : numbersList) {
            sum += number;
        }

        System.out.println("\nSoma dos números da lista: " + sum);

        // Invertendo a string aleatória gerada
        String reversedString = new StringBuilder(randomString).reverse().toString();
        System.out.println("String invertida: " + reversedString);
    }

    // Método para gerar uma string aleatória com um tamanho específico
    public static String generateRandomString(int length) {
        String characters = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789";
        StringBuilder stringBuilder = new StringBuilder();
        Random random = new Random();

        for (int i = 0; i < length; i++) {
            int randomIndex = random.nextInt(characters.length());
            stringBuilder.append(characters.charAt(randomIndex));
        }

        return stringBuilder.toString();
    }
}
